Class & Object
Blueprint (class) vs instance (object). __init__ is the constructor.
Code
class Car:
    # class variable - shared by all
    wheels = 4

    def __init__(self, brand, speed):
        # instance variables - unique per object
        self.brand = brand
        self.speed = speed

    def describe(self):
        return f"{self.brand} goes {self.speed}kmh"

# Creating objects (instances)
car1 = Car("Toyota", 120)
car2 = Car("BMW", 200)

print(car1.describe())  # Toyota goes 120kmh
print(car2.describe())  # BMW goes 200kmh
print(Car.wheels)       # 4 (class variable)
Key concepts

self — refers to the current instance. Always first parameter in instance methods.

__init__ — constructor, runs automatically when object is created.

Class variable — shared across all instances (wheels=4).

Instance variable — unique per object (brand, speed).

Interview tip: "A class is a blueprint, an object is a specific instance of that blueprint with its own state."
Inheritance
Child class inherits attributes and methods from parent class.
Code
class Animal:          # Parent
    def __init__(self, name):
        self.name = name

    def speak(self):
        return "..."

    def breathe(self):
        return "inhale/exhale"


class Dog(Animal):     # Child inherits Animal
    def __init__(self, name, breed):
        super().__init__(name)  # call parent init
        self.breed = breed

    def speak(self):        # override parent
        return f"{self.name} says Woof!"


dog = Dog("Rex", "Labrador")
print(dog.speak())    # Rex says Woof!
print(dog.breathe())  # inhale/exhale (from Animal)
print(isinstance(dog, Animal))  # True
Types of inheritance
# Single
class Dog(Animal): ...

# Multiple
class Amphibian(Land, Water): ...

# Multilevel
class Animal: ...
class Dog(Animal): ...
class Puppy(Dog): ...
Key methods
# Check inheritance
isinstance(dog, Animal)   # True
issubclass(Dog, Animal)   # True
Interview tip: "super() lets me call the parent class method without hardcoding the parent name — important in multiple inheritance."
Polymorphism
Same method name, different behaviour depending on the object.
Method overriding
class Shape:
    def area(self):
        return 0

class Circle(Shape):
    def __init__(self, r):
        self.r = r
    def area(self):
        return 3.14 * self.r ** 2

class Square(Shape):
    def __init__(self, s):
        self.s = s
    def area(self):
        return self.s ** 2

# Polymorphism in action
shapes = [Circle(5), Square(4)]
for s in shapes:
    print(s.area())  # 78.5, then 16
Duck typing (Python style)
# No inheritance needed!
class Dog:
    def speak(self): return "Woof"

class Cat:
    def speak(self): return "Meow"

def make_sound(animal):
    print(animal.speak())

make_sound(Dog())  # Woof
make_sound(Cat())  # Meow
# "If it has .speak(), it works!"
Interview tip: "Polymorphism means the same interface works for different types. In Python, duck typing means I don't even need inheritance — if the object has the method, it works."
Encapsulation
Hiding internal data. Control access using public, protected, private.
Code
class BankAccount:
    def __init__(self, balance):
        self.owner = "Supriya"   # public
        self._limit = 10000      # protected
        self.__balance = balance   # private

    def deposit(self, amount):
        if amount > 0:
            self.__balance += amount

    def get_balance(self):       # getter
        return self.__balance

acc = BankAccount(1000)
acc.deposit(500)
print(acc.get_balance())   # 1500
print(acc.owner)           # Supriya (public)
# acc.__balance → AttributeError!
Access levels
# public → accessible anywhere
self.name = "value"

# protected → convention: don't use outside
self._limit = 10000

# private → name mangled, truly hidden
self.__balance = 1000

# Python stores it as:
# _BankAccount__balance
print(acc._BankAccount__balance) # 1500
Interview tip: "Encapsulation protects object state. In Python, double underscore triggers name mangling — the attribute still exists but can't be accessed directly. It's convention-based, not enforced like Java."
Abstraction
Hide implementation details. Show only what's necessary using ABC.
Code
from abc import ABC, abstractmethod

class BasePage(ABC):   # abstract class
    def __init__(self, driver):
        self.driver = driver

    @abstractmethod
    def navigate(self):   # must implement
        pass

    @abstractmethod
    def is_loaded(self):  # must implement
        pass

    def click(self, el):  # shared method
        el.click()


class LoginPage(BasePage):
    def navigate(self):
        self.driver.get("/login")

    def is_loaded(self):
        return "login" in self.driver.url
Why it matters in SDET

Abstract base classes are exactly how Page Object Model works — BasePage defines the contract every page must follow.

# Can't instantiate abstract class!
page = BasePage(driver)
# → TypeError: Can't instantiate
#   abstract class BasePage

# Must use concrete child
page = LoginPage(driver)  # ✅
Interview tip: "Abstraction in my framework means BasePage defines what every page must have — navigate() and is_loaded() — but each page implements it differently. This enforces a contract across all page objects."
Magic Methods (Dunder)
Special methods with double underscores that Python calls automatically.
Common magic methods
class TestResult:
    def __init__(self, name, passed):
        self.name = name
        self.passed = passed

    def __str__(self):      # print(obj)
        status = "✅" if self.passed else "❌"
        return f"{status} {self.name}"

    def __repr__(self):     # debug repr
        return f"TestResult({self.name})"

    def __eq__(self, other):  # == operator
        return self.name == other.name

    def __len__(self):      # len(obj)
        return len(self.name)

r = TestResult("test_login", True)
print(r)        # ✅ test_login
print(len(r))   # 10
Quick reference
__init__   # constructor
__str__    # print(obj) friendly
__repr__   # debug representation
__len__    # len(obj)
__eq__     # obj1 == obj2
__lt__     # obj1 < obj2
__add__    # obj1 + obj2
__getitem__ # obj[key]
__contains__ # x in obj
__enter__  # with obj: (context mgr)
__exit__   # end of with block
Interview tip: "__str__ is for end users (print), __repr__ is for developers (debug). Always define __repr__ at minimum."
Properties (getter/setter)
Control attribute access with @property decorator — Pythonic encapsulation.
Code
class Employee:
    def __init__(self, name, salary):
        self.name = name
        self.__salary = salary  # private

    @property
    def salary(self):          # getter
        return self.__salary

    @salary.setter
    def salary(self, value):   # setter
        if value < 0:
            raise ValueError("No negative!")
        self.__salary = value

    @salary.deleter
    def salary(self):          # deleter
        del self.__salary

emp = Employee("Supriya", 50000)
print(emp.salary)     # 50000 (getter)
emp.salary = 60000    # setter called
emp.salary = -1       # ValueError!
Why use @property?

Access feels like a plain attribute but runs validation logic underneath. No need for get_salary() methods like Java.

# Without property (Java style)
emp.set_salary(60000)
emp.get_salary()

# With @property (Pythonic)
emp.salary = 60000   # clean!
emp.salary            # clean!
Interview tip: "@property gives you the clean syntax of direct attribute access while keeping validation logic hidden — best of both encapsulation and usability."
Class method vs Static method
classmethod gets the class, staticmethod gets nothing — just a utility.
Code
class DateHelper:
    date_format = "%Y-%m-%d"  # class var

    def __init__(self, date):
        self.date = date

    @classmethod
    def from_string(cls, date_str):
        # cls = DateHelper class itself
        return cls(date_str)   # alternate constructor

    @staticmethod
    def is_valid(date_str):
        # no self, no cls — pure utility
        return len(date_str) == 10

# classmethod - alternate constructor
d = DateHelper.from_string("2024-01-15")

# staticmethod - utility, no instance needed
print(DateHelper.is_valid("2024-01-15")) # True
Quick comparison
              self   cls   use case
──────────────────────────────────
instance method  ✅    ❌   normal method
@classmethod     ❌    ✅   alt constructor
@staticmethod    ❌    ❌   pure utility
SDET use case
class APIClient:
    @staticmethod
    def build_headers(token):
        return {
            "Authorization": f"Bearer {token}",
            "Content-Type": "application/json"
        }
Interview tip: "I use @staticmethod for pure utility functions like building headers or validating formats — they don't need instance or class state."
